```copy
// MENTE: Handlers de Evento
function toggleSection(sectionId) {
  const section = document.querySelector(`#${sectionId}`);
  section?.classList.toggle('active');
}

function setMode(mode) {
  localStorage.setItem('di_mode', mode);
  location.reload();
}

function toggleDrawer(drawerId) {
  document.querySelector(`#${drawerId}`)?.classList.toggle('open');
}

function minimizeWindow(windowId) {
  document.querySelector(`#${windowId}`)?.classList.add('minimized');
}

function cycleSessionWindow(windowId) {
  const window = document.querySelector(`#${windowId}`);
  window?.classList.toggle('collapsed');
}
```
