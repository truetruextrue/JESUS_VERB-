    (function() {
      window.addEventListener('load', () => {
        // Aguarda 1.5s para garantir que o ss5.js criou os iframes (YouTube/SoundCloud)
        setTimeout(() => {
          const mainProgress = document.getElementById('main-progress');
          const timeCurrent = document.getElementById('kodux-time-current');
          const timeTotal = document.getElementById('kodux-time-total');
          const localAudio = document.getElementById('local-audio');
          const footerClick = document.getElementById('footer-progress-click');
          const footerBar = document.getElementById('footer-progress-bar');

          let isDragging = false;
          let activeDuration = 0;
          let scWidgetCache = null;

          // Formatar segundos para MM:SS
          const formatTime = (seconds) => {
            if (!seconds || isNaN(seconds)) return "00:00";
            const m = Math.floor(seconds / 60);
            const s = Math.floor(seconds % 60);
            return `${m.toString().padStart(2, '0')}:${s.toString().padStart(2, '0')}`;
          };

          const updateTimeUI = (current, total) => {
            if (timeCurrent) timeCurrent.innerText = formatTime(current);
            if (timeTotal) timeTotal.innerText = formatTime(total);
          };

          // HACK: Interceptar a tentativa do ss5.js de atualizar a barra enquanto o utilizador a arrasta
          if (mainProgress) {
            const originalDescriptor = Object.getOwnPropertyDescriptor(HTMLInputElement.prototype, 'value');
            Object.defineProperty(mainProgress, 'value', {
              get: function() { return originalDescriptor.get.call(this); },
              set: function(val) {
                // Só permite que o script original mova a barra se o utilizador não estiver a mexer nela
                if (!isDragging) {
                  originalDescriptor.set.call(this, val);
                  if (footerBar) footerBar.style.width = `${val}%`;
                }
              }
            });
          }

          // Obter ou instanciar a API do SoundCloud
          const getSCWidget = () => {
            if (scWidgetCache) return scWidgetCache;
            const iframe = document.querySelector('#sc-container iframe');
            if (iframe && window.SC) scWidgetCache = SC.Widget(iframe);
            return scWidgetCache;
          };

          // Função que descobre quem está a tocar e extrai a duração/tempo atual
          const syncTimers = () => {
            // 1. Verificar Áudio Local
            if (localAudio && localAudio.duration && !localAudio.paused) {
              activeDuration = localAudio.duration;
              if (!isDragging) updateTimeUI(localAudio.currentTime, activeDuration);
              return;
            }

            // 2. Verificar YouTube (Tenta as variáveis globais mais comuns)
            const yt = window.ytPlayer || window.player;
            if (yt && typeof yt.getPlayerState === 'function') {
              // Se o estado for 1 (A tocar) ou se já tivermos a duração
              if (yt.getPlayerState() === 1 || yt.getDuration() > 0) {
                activeDuration = yt.getDuration();
                if (!isDragging) updateTimeUI(yt.getCurrentTime(), activeDuration);
                return;
              }
            }

            // 3. Verificar SoundCloud
            const sc = getSCWidget();
            if (sc) {
              sc.isPaused((paused) => {
                if (!paused) {
                  sc.getDuration((dur) => {
                    activeDuration = dur / 1000; // SC retorna milissegundos
                    sc.getPosition((pos) => {
                      if (!isDragging) updateTimeUI(pos / 1000, activeDuration);
                    });

            if (yt && typeof yt.seekTo === 'function') {
              const dur = yt.getDuration() || activeDuration;
              if (dur) yt.seekTo(pctDec * dur, true);
            }

            // Forçar Avanço no SoundCloud
            const sc = getSCWidget();
            if (sc) {
              sc.getDuration((dur) => {
                sc.seekTo(pctDec * dur); // SC aceita milissegundos
              });
            }

            };

                updateTimeUI(currentSec, activeDuration);
              }
