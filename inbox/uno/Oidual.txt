### 🪄 1. Comandos que se Personificam

```javascript
// A ferramenta que escuta e age
function parseComandoTexto(texto) {
  if (!texto?.startsWith('#')) return false;
  
  const [cmd, valor] = texto.substring(1).trim().split('=');
  if (!cmd || valor === undefined) return false;
  
  // Mapeia comandos de texto para objetos de comando
  const comandosMap = {
    'tema': { type: 'theme', value: valor.toUpperCase() },
    'audio': { type: 'audio', url: valor },
    'abrir': { type: 'open', target: valor || 'trinity' },
    // ...
  };
  
  const comando = comandosMap[cmd.toLowerCase()];
  if (comando) {
    executarComandoSimbiotico(comando);
    return true;
  }
  
  return false;
}
```

// A ferramenta que se integra sem dominar
function inserirTrinityContainer() {
  if (document.getElementById('trinityContainer')) return;
  
  const container = document.createElement('div');
  container.id = 'trinityContainer';
  container.className = 'trinity-hide';
  container.setAttribute('aria-hidden', 'true');
  container.innerHTML = `
    <iframe 
      id="trinityFrame" 
      src="trinity.html" 
      name="trinityFrame"
      style="width:90%;max-width:600px;height:80vh;border:none;background:transparent;"
      allowtransparency="true"
      sandbox="allow-scripts allow-same-origin allow-forms"
    ></iframe>
    <button id="trinityClose" aria-label="Fechar Trinity">✕</button>
  `;
  
  // Inserir no final do body para não interferir no fluxo existente
  document.body.appendChild(container);
  
  // Adicionar estilos mínimos apenas se não existirem
  if (!document.getElementById('trinity-styles')) {

### Fase 3: Conexão Inteligente com CallAI (5 minutos)
```javascript
// A ferramenta que amplifica sem substituir
if (typeof renderPaginatedResponse === 'function') {
  const originalRender = renderPaginatedResponse;
  window.renderPaginatedResponse = function(data) {
    // Processa comandos se existirem
    if (data?.commands && Array.isArray(data.commands)) {
      if (typeof window.DualSystem?.executarComandoSimbiotico === 'function') {
        data.commands.forEach(cmd => {
          window.DualSystem.executarComandoSimbiotico(cmd);
        });
      }
    }
    // Chama render original
    return originalRender.call(this, data);
  };
}
```
